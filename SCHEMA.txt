CREATE TABLE FST_SPECIALITATI (
  ID_SPECIALITATE NUMBER NOT NULL,
  NUME VARCHAR2(20),
  PRIMARY KEY (ID_SPECIALITATE) 
);

CREATE TABLE FST_SECTII (
  ID_SECTIE NUMBER NOT NULL,
  ID_SPITAL NUMBER,
  ID_SPECIALITATE NUMBER,
  ETAJ NUMBER,
  CORP VARCHAR(2),
  PRIMARY KEY (ID_SECTIE),
  FOREIGN KEY (ID_SPECIALITATE) REFERENCES FST_SPECIALITATI(ID_SPECIALITATE),
  FOREIGN KEY (ID_SPITAL) REFERENCES FST_SPITALE(ID_SPITAL)
);

CREATE TABLE FST_SALOANE (
  ID_SALON NUMBER NOT NULL,
  NR_PATURI NUMBER,
  ID_SECTIE NUMBER,
  PRIMARY KEY (ID_SALON),
  FOREIGN KEY(ID_SECTIE) REFERENCES FST_SECTII(ID_SECTIE)
);


CREATE TABLE FST_TARA (
  ID_TARA NUMBER NOT NULL,
  NUME VARCHAR2(12),
  PRIMARY KEY (ID_TARA) 
);

CREATE TABLE FST_LOCALITATE (
  ID_LOCALITATE NUMBER NOT NULL,
  ID_TARA NUMBER,
  NUME VARCHAR2(12),
  PRIMARY KEY (ID_LOCALITATE),
  FOREIGN KEY(ID_TARA) REFERENCES FST_TARA(ID_TARA)
);

CREATE TABLE FST_SPITALE (
  ID_SPITAL NUMBER NOT NULL,
  NUME VARCHAR2(20),
    ID_LOCALITATE NUMBER,
  ADRESA VARCHAR2(50),
  NR_TEL VARCHAR2(15),
  PRIMARY KEY (ID_SPITAL),
  FOREIGN KEY(ID_LOCALITATE) REFERENCES FST_LOCALITATE(ID_LOCALITATE)
);

CREATE TABLE FST_ANGAJATI (
  ID_ANGAJAT NUMBER  PRIMARY KEY  NOT NULL,
  NUME VARCHAR2(15),
  PRENUME VARCHAR2(10),
  SALARIU NUMBER,
  DATA_NASTERII DATE,
  ID_LOCALITATE NUMBER,
  FOREIGN KEY (ID_LOCALITATE) REFERENCES FST_LOCALITATE(ID_LOCALITATE)
);

CREATE TABLE FST_PACIENTI (
  ID_PACIENT NUMBER  NOT NULL,
  ID_LOCALITATE NUMBER,
  NUME VARCHAR2(15),
  PRENUME VARCHAR2(10),
  CNP NUMBER,
  NR_TEL NUMBER,
  ID_DOCTOR NUMBER,
  ID_MEDICAMENT NUMBER,
  PRIMARY KEY (ID_PACIENT),
  FOREIGN KEY (ID_MEDICAMENT) REFERENCES FST_MEDICAMENTE(ID_MEDICAMENT),
  FOREIGN KEY (ID_LOCALITATE) REFERENCES FST_LOCALITATE(ID_LOCALITATE)
  
 
);

CREATE TABLE FST_ASISTENTI (
  ID_ASISTENT NUMBER NOT NULL,
  ID_ANGAJAT NUMBER,
  ID_SALON NUMBER,
  PRIMARY KEY (ID_ASISTENT),
  FOREIGN KEY (ID_ANGAJAT) REFERENCES FST_ANGAJATI(ID_ANGAJAT),
  FOREIGN KEY (ID_SALON) REFERENCES FST_SALOANE(ID_SALON)
  
);

CREATE TABLE FST_CABINETE (
  ID_CABINET NUMBER NOT NULL,
  ID_SPECIALITATE NUMBER,
  ID_SECTIE NUMBER,
  PRIMARY KEY (ID_CABINET),
  FOREIGN KEY (ID_SPECIALITATE) REFERENCES FST_SPECIALITATI(ID_SPECIALITATE),
  FOREIGN KEY (ID_SECTIE) REFERENCES FST_SECTII(ID_SECTIE)
 
  
);

CREATE TABLE FST_DOCTORI (
  ID_DOCTOR NUMBER NOT NULL,
  GRAD VARCHAR2(10),
  ID_ANGAJAT NUMBER,
  ID_PACIENT NUMBER,
  PRIMARY KEY (ID_DOCTOR),
  FOREIGN KEY (ID_ANGAJAT) REFERENCES FST_ANGAJATI(ID_ANGAJAT)
);


CREATE TABLE FST_REPARTIZARE_SALOANE (

    ID_REPARTIZARE_SAL NUMBER NOT NULL,
    ID_ASISTENT NUMBER,
    ID_SALON NUMBER,
    PRIMARY KEY (ID_REPARTIZARE_SAL),
    FOREIGN KEY (ID_SALON) REFERENCES FST_SALOANE(ID_SALON),
    FOREIGN KEY (ID_ASISTENT) REFERENCES FST_ASISTENTI(ID_ASISTENT)
);
CREATE TABLE FST_REPARTIZARE_PACIENTI (
    ID_REPARTIZARE_PAC NUMBER NOT NULL,
    ID_PACIENT,
    ID_DOCTOR,
    PRIMARY KEY (ID_REPARTIZARE_PAC),
    FOREIGN KEY (ID_PACIENT) REFERENCES FST_PACIENTI(ID_PACIENT),
    FOREIGN KEY (ID_DOCTOR) REFERENCES FST_DOCTORI(ID_DOCTOR)
    );
CREATE TABLE FST_SPECIALITATI_DOCTORI (
  ID_SPECIALITATE NUMBER ,
  ID_DOCTOR NUMBER,
  ID_SP NUMBER NOT NULL,
  PRIMARY KEY (ID_SP),
  FOREIGN KEY (ID_SPECIALITATE) REFERENCES FST_SPECIALITATI(ID_SPECIALITATE),
  FOREIGN KEY (ID_DOCTOR) REFERENCES FST_DOCTORI(ID_DOCTOR)
);
